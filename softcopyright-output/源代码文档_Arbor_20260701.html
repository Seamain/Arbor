    let state = app.state::<WatcherState>();

    let git_dir = std::path::Path::new(&path).join(".git");
    let mut debouncer = new_debouncer(
        Duration::from_millis(500),
        move |result: DebounceEventResult| {

                    !e.path.starts_with(&git_dir)

    debouncer
        .watcher()

    Ok(())

fn safe_join(repo_path: &str, file_path: &str) -> Result<PathBuf, String> {
    let base = Path::new(repo_path)
        .canonicalize()
        .map_err(|e| format!("Cannot resolve repo path: {}", e))?;
    let full = base.join(file_path);
    if let Ok(canon) = full.canonicalize() {
        if !canon.starts_with(&base) {
            return Err(format!(
                "Path '{}' is outside the repository directory.", file_path
            ));
        }
        return Ok(canon);
    }
    let parent = full
        .parent()
        .unwrap_or_else(|| Path::new(repo_path));
    let canon_parent = parent
        .canonicalize()
        .map_err(|e| format!("Cannot resolve parent directory: {}", e))?;
    if !canon_parent.starts_with(&base) {
        return Err(format!(
            "Path '{}' is outside the repository directory.", file_path
        ));
    }
    Ok(full)
}

fn relative_time(secs: i64) -> String {
    let secs = secs.max(0) as u64;
    if secs < 60 {
        return "just now".to_string();
    }
    let mins = secs / 60;
    if mins < 60 {
        return format!("{} minute{} ago", mins, if mins == 1 { "" } else { "s" });
    }
    let hours = mins / 60;
    if hours < 24 {
        return format!("{} hour{} ago", hours, if hours == 1 { "" } else { "s" });
    }
    let days = hours / 24;
    if days < 30 {
        return format!("{} day{} ago", days, if days == 1 { "" } else { "s" });
    }
    let months = days / 30;
    if months < 12 {
        return format!("{} month{} ago", months, if months == 1 { "" } else { "s" });
    }
    let years = months / 12;
    format!("{} year{} ago", years, if years == 1 { "" } else { "s" })
}
